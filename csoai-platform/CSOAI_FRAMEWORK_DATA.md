# DATABASE SCHEMA (Future)

```sql
-- Frameworks table
CREATE TABLE frameworks (
  id VARCHAR PRIMARY KEY,
  name VARCHAR,
  full_name VARCHAR,
  type VARCHAR,
  region VARCHAR,
  country VARCHAR,
  status VARCHAR,
  mandatory BOOLEAN,
  enforcement_date DATE,
  description TEXT,
  url VARCHAR
);

-- Controls table
CREATE TABLE controls (
  id VARCHAR PRIMARY KEY,
  name VARCHAR,
  description TEXT,
  category VARCHAR,
  severity VARCHAR
);

-- Control-Framework mapping
CREATE TABLE control_framework_map (
  control_id VARCHAR,
  framework_id VARCHAR,
  framework_requirement VARCHAR,
  PRIMARY KEY (control_id, framework_id)
);

-- Company compliance
CREATE TABLE company_compliance (
  company_id VARCHAR,
  control_id VARCHAR,
  status VARCHAR,
  last_checked DATE,
  evidence TEXT,
  PRIMARY KEY (company_id, control_id)
);
```
